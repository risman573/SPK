-- PROVINSI --
UPDATE data_full AS df
JOIN m_provinsi AS mp 
    ON TRIM(LOWER(df.provinsi)) = TRIM(LOWER(mp.nama))
SET df.id_provinsi = mp.id

SELECT id_provinsi, provinsi FROM `data_full` WHERE id_provinsi='' GROUP BY provinsi ORDER BY `data_full`.`id_provinsi` ASC

-- KOTA --
UPDATE data_full AS df
JOIN m_kota AS mk 
    ON TRIM(LOWER(
        CASE 
            WHEN LEFT(mk.nama, 5) = 'kota ' THEN SUBSTRING(mk.nama, 6)
            WHEN LEFT(mk.nama, 10) = 'kabupaten ' THEN SUBSTRING(mk.nama, 11)
            ELSE mk.nama
        END
    )) = TRIM(LOWER(df.kota))
SET df.id_kota = mk.id
